import axios from 'axios'
import authHeader from "./auth-header";

const PLAN_BASE_REST_API_URL = 'https://8080-ddbdacaccaaebbfaaecebafeebbfdeebfce.examlyiopb.examly.io/admin';

class MonthlyService{

    getAllPlans(){
        return axios.get(PLAN_BASE_REST_API_URL+'/getAllPlan', { headers: authHeader() })
    }

    createPlan(plan){
        return axios.post(PLAN_BASE_REST_API_URL +'/addPlan', plan, { headers: authHeader() })
    }
    
    getPlanById(planId){
        return axios.get(PLAN_BASE_REST_API_URL + '/getPlan/' + planId, { headers: authHeader() });
    }

    updatePlan(planId, plan){
        return axios.put(PLAN_BASE_REST_API_URL + '/editPlan/' + planId, plan, { headers: authHeader() });
    }

    deletePlan(planId){
        return axios.delete(PLAN_BASE_REST_API_URL + '/deletePlan/' + planId, { headers: authHeader() });
    }
}

export default new MonthlyService();
