import axios from 'axios'
import authHeader from "./auth-header";

const ADDON_BASE_REST_API_URL = 'https://8080-ddbdacaccaaebbfaaecebafeebbfdeebfce.examlyiopb.examly.io/admin';

class AddonService{

    getAllAddons(){
        return axios.get(ADDON_BASE_REST_API_URL+'/getAddon', { headers: authHeader() })
    }

    createAddon(addon){
        return axios.post(ADDON_BASE_REST_API_URL +'/addAddon', addon, { headers: authHeader() })
    }

    getAddonById(addonId){
        return axios.get(ADDON_BASE_REST_API_URL + '/getAddon/' + addonId, { headers: authHeader() });
    }

    updateAddon(addonId, addon){
        return axios.put(ADDON_BASE_REST_API_URL + '/editAddon/' + addonId, addon, { headers: authHeader() });
    }

    deleteAddon(addonId){
        return axios.delete(ADDON_BASE_REST_API_URL + '/deleteAddon/' + addonId, { headers: authHeader() });
    }
}

export default new AddonService();
