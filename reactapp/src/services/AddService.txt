import axios from 'axios'
import authHeader from "./auth-header";

const ADDON_BASE_REST_API_URL = 'https://8080-ddbdacaccaaebbfaaecebafeebbfdeebfce.examlyiopb.examly.io/user';

class AddService{

    getAllAddons(){
        return axios.get(ADDON_BASE_REST_API_URL+'/getAddon', { headers: authHeader() })
    }

    getAddonById(addonId){
        return axios.get(ADDON_BASE_REST_API_URL + '/getAddon/' + addonId, { headers: authHeader() });
    }

    createRecharge(plan){
        return axios.post(ADDON_BASE_REST_API_URL +'/addRecharge', plan, { headers: authHeader() })
    }

    getAllRecharge(){
        return axios.get(ADDON_BASE_REST_API_URL+'/getRecharge', { headers: authHeader() })
    }
}

export default new AddService();
